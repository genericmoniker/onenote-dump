import numpy as np


def pagerank(M, num_iterations=100, d=0.85):

N = M.shape[1]

v = np.random.rand(N, 1)

v = v / np.linalg.norm(v, 1)

iteration = 0

while iteration < num_iterations:

iteration += 1

v = d * np.matmul(M, v) + (1 - d) / N

return v
